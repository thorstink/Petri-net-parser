\documentclass[border = 0.2cm]{standalone}

% Required packages and libraries
\usepackage{tikz}
\usetikzlibrary{petri,positioning}

\begin{document}

\begin{tikzpicture}

% Place & transitions
\node[  place,
        label=below:$P_1$,
        tokens=1
    ] (place1) at (0,0) {};
\node[  transition,
        label=below:$T_1$,
        right=of place1,
        minimum width=2mm,
        minimum height=8mm,
        fill=black,
    ] (trans1) {};
\node[  place,
        label=below:$P_2$,
        right=of trans1
    ] (place2) {};

% Arcs
\draw[thick] (place1) edge[post] (trans1)
    (trans1) edge[post] (place2);
\end{tikzpicture}
